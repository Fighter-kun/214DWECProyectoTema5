4. Crea una web con contenido. Crea una aplicación que pida una palabra al usuario y marque dentro del texto, todas las apariciones de esa palabra. Para ello se aconseja la creación de dos funciones. a. Una para marcar las palabras (Pistas: innerHTML, replace, expresiones regulares) b. Otra recursiva que recorra el DOM:


                            function searchWords() {
                            let words = document.getElementById("words").value;
                            let expr = new RegExp(words, 'gi');


                            document.body.innerHTML = document.body.innerHTML.replace(/< strong>/g, "");
                            document.body.innerHTML = document.body.innerHTML.replace(/<\/strong>/g, "");


                            function recorrerDOM(elemento) {
                              let expr = new RegExp(words, 'gi');
                              // Imprimir la etiqueta del elemento y su nivel en la consol

                              for (let i = 0; i < elemento.children.length; i++) {
                                // Llamada recursiva para cada hijo
                                recorrerDOM(elemento.children[i]);
                              }

                              if (elemento.textContent.includes(words)) {
                                elemento.innerHTML = elemento.innerHTML.replace(expr, `< strong>${words}< /strong>`);
                              }
                            }
                            // Llamar a la función con el elemento raíz del DOM (puedes ajustar esto según tus necesidades)
                            recorrerDOM(document.body);

                          }
