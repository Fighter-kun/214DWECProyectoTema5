1. Crea una página que muestre una imagen u otra en función del tamaño de la pantalla y los px de alto y de ancho que ocupa la barra de tareas del usuario:


                            function updateImageVisibility() {
                            var largeImage = document.getElementById("largeImage");
                            var mediumImage = document.getElementById("mediumImage");
                            var smallImage = document.getElementById("smallImage");

                            // Obtener el ancho de la ventana
                            var windowWidth = window.innerWidth || document.documentElement.clientWidth || document.body.clientWidth;

                            // Determinar el tamaño de la imagen según el ancho de la ventana
                            if (windowWidth >= 1200) {
                                largeImage.style.display = "block";
                                mediumImage.style.display = "none";
                                smallImage.style.display = "none";
                            } else if (windowWidth >= 800) {
                                largeImage.style.display = "none";
                                mediumImage.style.display = "block";
                                smallImage.style.display = "none";
                            } else if (windowWidth >= 600) {
                                largeImage.style.display = "none";
                                mediumImage.style.display = "none";
                                smallImage.style.display = "block";
                            } else {
                                // Si ninguna de las condiciones anteriores es verdadera
                                largeImage.style.display = "none";
                                mediumImage.style.display = "none";
                                smallImage.style.display = "none";
                            }

                        }

                        // Ejecutar la función al cargar la página y al cambiar el tamaño de la ventana
                        document.addEventListener("DOMContentLoaded", updateImageVisibility);
                        window.addEventListener("resize", updateImageVisibility);
                        // Esperar a que el DOM se cargue antes de ejecutar el script
                        document.addEventListener("DOMContentLoaded", function () {
                          // Obtengo el tamaño completo de la pantalla
                          var screenHeight = screen.height - screen.availHeight;
                          var screenWidth = window.screen.width;

                          // Aqui obtengo la altura sin la barra de tareas
                          var windowHeight = window.innerHeight || document.documentElement.clientHeight || document.body.clientHeight;

                          let tamañoBarraDeTareas = document.getElementById("tamañoBarraDeTareas");
                          tamañoBarraDeTareas.innerHTML = "< div>Resolución de la barra de tareas "+ screenHeight + " X " + screenWidth + " px< /div>";
                        });
